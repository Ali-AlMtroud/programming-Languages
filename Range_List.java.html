package sample;

import java.util.ArrayList;

public class Range_List {
    ArrayList<String>animes;
    ArrayList<String>position;
    Range_List()
    {
        this.animes=new ArrayList<String>();
        this.position=new ArrayList<String>();
    }

    public ArrayList<String> getAnimes() {
        return animes;
    }

    public void setAnimes(ArrayList<String> animes) {
        this.animes = animes;
    }

    public ArrayList<String> getPosition() {
        return position;
    }

    public void setPosition(ArrayList<String> position) {
        this.position = position;
    }
}
