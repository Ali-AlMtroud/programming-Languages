    public void Read_log_file1(String type,String num) {
        a.clear();
        if(type.length()==0 && num.length()==0)
        {
            a.clear();
            a.add("this file is not found");
        }
        try {

            String jj;

            BufferedReader file=new BufferedReader(new FileReader(type+"\\"+num));
            while ((jj=file.readLine())!=null) {
                // System.out.println(jj);

                a.add(jj);
            }


            file.close();
        } catch (Exception e) {
            e.printStackTrace();
        }
    }

    public void movement(Rectangle r,int x,int y,int xx,int yy,Image p)
    {
       r.setHeight(50);
        r.setWidth(50);
        r.setFill(new ImagePattern(p));
         r.setTranslateX(xx);
         r.setTranslateY(yy);
        // بالعرض و الطول rectangle هنا قنما بتحديد درجة إنحناء زوايا الكائن
        r.setArcWidth(30);
        r.setArcHeight(30);
        TranslateTransition translateTransition = new TranslateTransition();
        // عندما يتم تشغيله translateTransition هنا قمنا بتحديد المدة التي سيتنفذ خلالها التأثير الذي يمثل الكائن
        translateTransition.setDuration(Duration.seconds(80));

        //  سيتم تطبيقه على الكائن translateTransition هنا حددنا أن التأثير الذي سنحصل عليه عند تشغيل الكائن
        translateTransition.setNode(r);
        // هو التحرك للأسفل بنسبة 80 بيكسل translateTransition هنا قلنا أن التأثير الذي سنحصل عليه عند تشغيل الكائن
