package sample;

import java.io.File;
import java.io.FileNotFoundException;
import java.io.IOException;
import java.util.Scanner;

public class River {
    int x;
    int y;

    public River() throws FileNotFoundException {


        try {
            File myObj = new File("River.txt");

            Scanner myReader = new Scanner(myObj);
            if (myReader.hasNextInt()) {
                this.x = myReader.nextInt();
                this.y = myReader.nextInt();
            }
            myReader.close();
        } catch (IOException e) {
            System.out.println("An error occurred.");
            e.printStackTrace();
        }
    }


    public int getX() {
        return x;
    }

    public int getY() {
        return y;
    }

}
