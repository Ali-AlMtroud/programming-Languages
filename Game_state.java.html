package sample;

public enum Game_state {

    Attacker_is_Winner("Attacker is Win"),
    Attacker_is_loser("Attacker is lose"),
    Running("Game is continue");

    private String value;

    Game_state(String s) {
        this.value = s;
    }


    public String getValue() {
        return value;
    }

}
