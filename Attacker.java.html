    synchronized ArrayList<Defender.Canon> check_Defender_cannon(ArrayList<String> b, ArrayList<String> bb) {
        ArrayList<Defender.Canon> Canon_health = new ArrayList<Defender.Canon>();
        if (bb.size() == 0)
            return null;

        try {


            for (int i = 0; i < b.size(); i++) {

                if (b.get(i).equals("&")) {


                    if (bb.get(i).length() == 2) {

                        int t = ((int) bb.get(i).charAt(0) - 48);
                        int tt = ((int) bb.get(i).charAt(1) - 48);

                        for (Defender.Canon Canon : s4) {
                            if (Canon.getX() == t && Canon.getY() == tt) {
                                Canon_health.add(Canon);
                            }
                        }
                    } else {

                        int t = 10 * ((int) bb.get(i).charAt(0) - 48);

                        int tt = ((int) bb.get(i).charAt(1) - 48);

                        int k = t + tt;

                        int ttt = 10 * ((int) bb.get(i).charAt(2) - 48);

                        int tttt = ((int) bb.get(i).charAt(3) - 48);
                        int kk = ttt + tttt;

                        for (Defender.Canon Canon : s4) {

                            if (Canon.getX() == k && Canon.getY() == kk) {

                                Canon_health.add(Canon);
                            }
                        }

                    }

                }

            }
        } catch (StringIndexOutOfBoundsException e) {
            e.printStackTrace();
        }
        return Canon_health;
    }

        void move() {

            Cell cell = new Cell(this.x, this.y);
            next_move(cell);
            this.x = cell.getX();
            this.y = cell.getY();
            if (!(arena.Getcell(cell.getX(), cell.getY()).equals("/") && !(arena.Getcell(this.x, this.y).equals("*"))))
                Game.arena.setArena(this.x, this.y, "T");
            try {
                Thread.sleep(500);
            } catch (InterruptedException e) {
                e.printStackTrace();
            }


        }

                Game.arena.setArena(this.x, this.y, "B");


        }
