    synchronized ArrayList<Attacker.ATank> check_Attack_Tanks(ArrayList<String> b, ArrayList<String> bb) {

        if (bb.size() == 0)
            return null;
        ArrayList<Attacker.ATank> Tanks_health = new ArrayList<Attacker.ATank>();

        for (int i = 0; i < b.size(); i++) {
            if (b.get(i).equals("T")) {

                if (bb.get(i).length() == 2) {

                    int t = bb.get(i).charAt(0) - 48;

                    int tt = bb.get(i).charAt(1) - 48;

                    for (Attacker.ATank tank : s1) {

                        if (tank.getX() == t && tank.getY() == tt) {

                            Tanks_health.add(tank);
                        }
                    }
                } else {

                    int t = 10 * ((int) bb.get(i).charAt(0) - 48);

                    int tt = ((int) bb.get(i).charAt(1) - 48);

                    int k = t + tt;

                    int ttt = 10 * ((int) bb.get(i).charAt(2) - 48);
                    int tttt = ((int) bb.get(i).charAt(3) - 48);
                    int kk = ttt + tttt;

                    for (Attacker.ATank tank : s1) {

                        if (tank.getX() == k && tank.getY() == kk) {

                            Tanks_health.add(tank);
                        }
                    }

                }

            }

        }
        return Tanks_health;

    }

    }

    public class Canon implements Runnable {
        int x;
        int y;
        int Health = 100;
        int armor = 1;
        final int damage = 50;
        int num_Devender;
        int num_Canon;

        public int getArmor() {
            return armor;
        }

        public Canon(int f, int num_Devender, int num_Canon) {
            if (MoneyD - 200 < 0) {
                System.out.println("your money isn't enogh");
                System.gc();
                return;
            } else {
                MoneyD -= 200;
                try {
                    File myObj = new File("Canon.txt");
                    Scanner myReader = new Scanner(myObj);

                    while (myReader.hasNextInt()) {

                        canonlist.add(myReader.nextInt());


                    }


                    myReader.close();
                } catch (IOException e) {
                    System.out.println("An error occurred.");
                    e.printStackTrace();
                }
                this.x = canonlist.get(f);

                this.y = canonlist.get(++f);

                arena.setArena(this.x, this.y, "&");


                s4.add(this);
                this.num_Devender = num_Devender;
                this.num_Canon = num_Canon;
                canonlist.clear();
            }
        }

        public int getX() {
            return x;
        }


        public int getY() {
            return y;
        }


        public int getHealth() {
            return Health;
        }

        public void setHealth(int health) {
            Health = health;
        }

        public int getDamage() {
            return damage;
        }

        void set_canon_in_arena() {

            Game.arena.setArena(this.x, this.y, "&");
        }


        void Attack_Range_canon(int random) {
            if (this.getHealth() <= 0)
                return;
            String info = "";
            Range_List List = Defender.Range(this.x, this.y, 2);
            info += "Cannon in ( " + Integer.toString(this.x) + ", " + Integer.toString(this.y) + ") and Health: " + Integer.toString(this.getHealth());
            if (List.getAnimes().size() == 0 || List.getPosition().size() == 0) {
                Write_log_file(Game.arena.file_name_Cannon(num_Devender, num_Canon), info);
                return;
            }
            if (List.getAnimes().contains("/")) {
                info += " and attack Bridge ";
                Write_log_file(Game.arena.file_name_Cannon(this.num_Devender, this.num_Canon), info);
                check_Attack_Bridge(List.getAnimes(), List.getPosition());
                return;
            }
            ArrayList<Attacker.ATank> check = check_Attack_Tanks(List.getAnimes(), List.getPosition());
            ArrayList<Attacker.Rsolider> check1 = check_Attack_r_solider(List.getAnimes(), List.getPosition());
            ArrayList<Attacker.Ssolider> check2 = check_Attack_s_solider(List.getAnimes(), List.getPosition());
            ArrayList<Attacker.Bomb> check4 = check_Attack_r_Bomb(List.getAnimes(), List.getPosition());
            if (random == 1) {
                Attacker.ATank kk = check.get(0);
                Attacker.Rsolider jj = check1.get(0);
                Attacker.Ssolider ll = check2.get(0);
                Attacker.Bomb mm = check4.get(0);
                if (mm != null) {
                    info += Set_Damage_Bomb(mm, Player.s8, info, this.damage);
                    Write_log_file(Game.arena.file_name_Cannon(this.num_Devender, this.num_Canon), info);
                    return;
                } else if (kk != null) {
                    info += Set_Damage_Atank(kk, Player.s1, info, this.damage);
                    Write_log_file(Game.arena.file_name_Cannon(this.num_Devender, this.num_Canon), info);
                    return;

                } else if (jj != null) {
                    info += Set_Damage_Rsolider(jj, Player.s3, info, this.damage);
                    Write_log_file(Game.arena.file_name_Cannon(this.num_Devender, this.num_Canon), info);
                    return;
                } else if (ll != null) {

                    info += Set_Damage_Solider(ll, Player.s2, info, this.damage);
                    Write_log_file(Game.arena.file_name_Cannon(this.num_Devender, this.num_Canon), info);
                    return;
                }
            } else if (random == 0) {
                Attacker.ATank kk = check_Attack_Tanks_less_health(check);
                Attacker.Rsolider jj = check_Attack_Rsolider_less_health(check1);
                Attacker.Ssolider ll = check_Attack_Ssolider_less_health(check2);
                Attacker.Bomb mm = check_Attack_Bomb_less_health(check4);
                if (mm != null) {
                    info += Set_Damage_Bomb(mm, Player.s8, info, this.damage);
                    Write_log_file(Game.arena.file_name_Cannon(this.num_Devender, this.num_Canon), info);
                    return;
                } else if (kk != null) {
                    info += Set_Damage_Atank(kk, Player.s1, info, this.damage);
                    Write_log_file(Game.arena.file_name_Cannon(this.num_Devender, this.num_Canon), info);
                    return;

                } else if (jj != null) {
                    info += Set_Damage_Rsolider(jj, Player.s3, info, this.damage);
                    Write_log_file(Game.arena.file_name_Cannon(this.num_Devender, this.num_Canon), info);
                    return;
                } else if (ll != null) {
                    info += Set_Damage_Solider(ll, Player.s2, info, this.damage);
                    Write_log_file(Game.arena.file_name_Cannon(this.num_Devender, this.num_Canon), info);
                    return;
                }
            }

        }


        @Override
        public void run() {
            while (Game.endGame() == Game_state.Running) {
                this.Attack_Range_canon(0);
                //   Game.Sleep_Unit(0.05);
                try {
                    Thread.sleep(500);
                } catch (InterruptedException e) {
                    e.printStackTrace();
                }
            }
            Thread.currentThread().stop();
        }
    }
