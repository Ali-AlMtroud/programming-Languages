    int X_Base() {
        int pos_x = 0, pos_y = 0;
        try {
            File myObj = new File("Base.txt");

            Scanner myReader = new Scanner(myObj);
            if (myReader.hasNextInt()) {
                pos_x = myReader.nextInt();
                pos_y = myReader.nextInt();

            }
            myReader.close();
        } catch (IOException e) {
            System.out.println("An error occurred.");
            e.printStackTrace();
        }
        return pos_x;
    }

    public String file_name_Atank(int num_attacker, int num_Atank) {
