package sample;

import java.io.File;
import java.io.FileNotFoundException;
import java.io.IOException;
import java.util.Scanner;

public class Bridge {
    int x;
    int y;
    double healtBridge = 100;

    public Bridge() throws FileNotFoundException {


        try {
            File myObj = new File("Bridge.txt");

            Scanner myReader = new Scanner(myObj);
            if (myReader.hasNextInt()) {
                this.x = myReader.nextInt();
                this.y = myReader.nextInt();
            }
            myReader.close();
        } catch (IOException e) {
            System.out.println("An error occurred.");
            e.printStackTrace();
        }
    }


    public void setHealtBridge(double healtBridge) {
        this.healtBridge = healtBridge;
    }

    public int getX() {
        return x;
    }

    public int getY() {
        return y;
    }

    public double getHealtBase() {
        return healtBridge;
    }
}
