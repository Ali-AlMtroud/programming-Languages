package sample;

import java.io.FileNotFoundException;

public class Game implements Runnable {
    protected static Arena arena;
    static Base base;
    Valley valley;
    River river;
    Bridge bridge;
    static Game_state state;

    public Game() {
        arena = new Arena();
    }

    public static void Sleep_Unit(double second) {
        double start = System.currentTimeMillis();
        double end = start + (1000 * second);
        while (System.currentTimeMillis() <= end) {

        }

    }

    public void Star_New_Game() throws FileNotFoundException {
        this.base = new Base();
        this.river = new River();
        this.valley = new Valley();
        this.bridge = new Bridge();
        arena.setArena(base.getX(), base.getY(), "#");
        arena.setArena(river.getX(), river.getY(), "*");
        arena.setArena(river.getX() + 1, (river.getY()), "*");
        arena.setArena(river.getX() + 2, (river.getY()), "*");
        arena.setArena(river.getX() + 3, (river.getY()), "*");

        arena.setArena(valley.getX(), valley.getY(), "+");
        arena.setArena(valley.getX() + 1, valley.getY(), "+");
        arena.setArena(valley.getX(), valley.getY() + 1, "+");
        arena.setArena((valley.getX()) + 1, (valley.getY()) + 1, "+");
        arena.setArena(bridge.getX(), bridge.getY(), "/");
        arena.setArena(bridge.getX() + 1, bridge.getY() + 1, "/");
        System.out.print("");

    }

    static boolean search_Attacker() {
        {
            if ((Player.s1.size() == 0) && (Player.s2.size() == 0) && (Player.s3.size() == 0) && (Player.s8.size() == 0) && (Player.s7.size() == 0))
                return true;
        }
        return false;
    }

    public static Game_state endGame() {

        if (base.getHealtBase() <= 0.0) {

            state = Game_state.Attacker_is_Winner;
            return state;
        } else if (base.getHealtBase() > 0 && (search_Attacker())) {
            state = Game_state.Attacker_is_loser;
            return state;
        }
        ///////////////////////////////////////////////////////////////////////////////////////////
        else if (base.getHealtBase() > 0 && (!search_Attacker())) {
            state = Game_state.Running;
            return state;

        }
        else {
            return null;
        }
    }


    @Override

    public void run() {
        try {
            this.Star_New_Game();
            Game.arena.printer();
            System.out.println();
            System.out.println();


        } catch (FileNotFoundException e) {
            e.printStackTrace();
        }
    }


}
